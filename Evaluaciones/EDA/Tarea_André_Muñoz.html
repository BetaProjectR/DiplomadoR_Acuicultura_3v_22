6. Identifica si existen errores, datos faltantes o valores atpicos (4 puntos).

contar_outliers <- function(column){

Q1 <- quantile(column, 0.25)
Q3 <- quantile(column, 0.75)
iqr <- IQR(column)

outlier_sup = (iqr * 1.5) + Q3
outlier_inf = Q1 - (iqr * 1.5) 

count_sup <- sum(column > outlier_sup)
count_inf <- sum(column < outlier_inf)

total_outliers <- round(count_sup + count_inf,1)
perc_outliers <- round(total_outliers/length(column),3)

salida = c(total_outliers, perc_outliers)
return(salida)

}
